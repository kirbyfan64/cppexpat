00122     void ParserBase::parse(std::ifstream& f, int sz=bufsize)
00123     {
00124         XML_Status s;
00125         char* buf;
00126         if (f.eof()) return;
00127         for (;;)
00128         {
00129             buf = static_cast<char*>(XML_GetBuffer(this->p, sz));
00130             if (!buf) throw std::bad_alloc{};
00131             f.read(buf, sz);
00132             s = XML_ParseBuffer(this->p, f.gcount(), f.eof());
00133             if (!s) throw XMLError(this->p);
00134             if (f.eof()) break;
00135         }
00136     }
