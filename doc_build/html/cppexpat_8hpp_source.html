00125     void ParserBase::parse(std::istream& in, int sz=bufsize)
00126     {
00127         XML_Status s;
00128         std::unique_ptr<char[]> buf{new char[sz]};
00129         if (in.eof()) return;
00130         for (;;)
00131         {
00132             in.read(buf.get(), sz);
00133             s = XML_Parse(this->p, buf.get(), in.gcount(), in.eof());
00134             if (!s) throw XMLError(this->p);
00135             if (in.eof()) break;
00136         }
00137     }
